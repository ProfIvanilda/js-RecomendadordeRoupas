let campoTempo; let campoEvento; let campoPeriodo; function setup() { createCanvas(1500, 400); createElement("h2", "Recomendador de Roupa para Mulheres"); createSpan("Qual é o Tempo? (Frio ou Calor):"); campoTempo = createInput("Calor"); createSpan("Qual é o tipo do Evento? (Festa, Trabalho ou Outros):"); campoEvento = createInput("Festa"); createSpan(" Qual é o Periodo do Evento? (dia ou Noite): "); campoPeriodo = createInput("Dia"); } function draw() { background("blue"); let tempo = campoTempo.value().toLowerCase(); let evento = campoEvento.value().toLowerCase(); let periodo = campoPeriodo.value().toLowerCase(); let recomendacao = geraRecomendacao(tempo, evento, periodo); fill(255); // Cor branca textAlign(CENTER, CENTER); textSize(25); text(recomendacao, width / 2, height / 2); } function geraRecomendacao(tempo, evento, periodo) { if (tempo === "calor") { if (evento === "festa") { if (periodo === "dia") { return "Use Vestido longo Frente Única Estampado"; } else { return "Use Vestido na altura do joelho de cor única"; } } else if (evento === "trabalho") { return "Use Saia ou calça de alfaiataria com uma blusa ou camisa"; } } else if (tempo === "frio") { if (evento === "festa") { if (periodo === "dia") { return "Use um vestido midi com meia-calça e um casaco leve"; } else { return "Use um vestido de manga longa e um casaco elegante"; } } else if (evento === "trabalho") { return "Use calça de alfaiataria, uma blusa de manga longa e um blazer"; } else { return "Use Calça Djeans com Blusa de lã e botas"; } } return "Use Calça Jeans e Camiseta"; }
